program futar;
type futaros=record
       nap: integer;
       fuvarsz: integer;
       km: integer;
     end;
var i,j,k,l,n,kezd,veg,szabadnap,maxi,
    napmin,napmax,sum,ut,ar:integer;
    csere: futaros;
    T:array[1..1000] of futaros;
    f:text;
function ut_ar(km:integer):integer;
begin
    case km of
    1..2: ut_ar:=500;
    3..5: ut_ar:=700;
    6..10: ut_ar:=900;
    11..20: ut_ar:=1400;
    21..30: ut_ar:=2000;
    else ut_ar:= -1;
  end;
end;
